CREATE TABLE pokemon(
    id INTEGER NOT NULL, 
    name TEXT NOT NULL,
    type1 TEXT NOT NULL,
    type2 TEXT,
    ability1 TEXT NOT NULL,
    ability2 TEXT,
    abilityH TEXT,
    hp INTEGER NOT NULL,
    attack INTEGER NOT NULL,
    defense INTEGER NOT NULL,
    spattack INTEGER NOT NULL,
    spdefense INTEGER NOT NULL,
    speed INTEGER NOT NULL,
    total INTEGER NOT NULL,
    wgt REAL NOT NULL,
    hgt REAL NOT NULL,
    classification TEXT NOT NULL,
    percentMale REAL NOT NULL,
    percentFemale REAL NOT NULL,
    preEvolution TEXT,
    eggGroup1 TEXT NOT NULL,
    eggGroup2 TEXT
);
